txt <- read.csv('주요선수별성적-2013년.csv',stringsAsFactor=F)
baseball <- txt %>%
  mutate(OPS= 출루율 + 장타율) %>%
  mutate(연봉대비OPS = (OPS/연봉)*100) %>%
  mutate(평균출루율 = sum(출루율)/25)

연봉대비출루율

palete <- c(brewer.pal(12, 'Set3'), 
            brewer.pal(12, 'Set3'),'#FFFFFF') # brewer.pal.info # palete 정보
